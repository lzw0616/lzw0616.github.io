 // 方法一：使用数组
        function judgeDay() {
            var sum = 0; //要返回的结果
            var Date1 = new Date();
            var arr = [31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31];
            var year = Date1.getFullYear();
            var month = Date1.getMonth();
            var day = Date1.getDate();
            for (var i = 0; i < month + 1; i++) {
                sum += arr[i];
            }
            if (judgeYear(year)) {
                sum++;
            }
            sum += day;
            return sum;
        }


        function judgeYear(year) {
            if ((year % 4 == 0 && year % 100 != 0) || year % 400 == 0) {
                return true;
            }
            return false;
        }
